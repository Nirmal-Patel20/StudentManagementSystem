cmake_minimum_required(VERSION 3.23)
project(StudentManagementSystem)

# Set C++ standard
set(CMAKE_CXX_STANDARD 23)

# Include path
include_directories(include)

# Collect all .cpp files from src/
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS src/*.cpp)

# Collect all header files from include/ (not strictly needed for compilation but useful for IDEs)
file(GLOB_RECURSE HEADERS CONFIGURE_DEPENDS include/*.hpp include/*.h)

# Create the executable
add_executable(StudentManagementSystem ${SOURCES} ${HEADERS})
